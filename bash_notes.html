  • Find intersection between lines in two files 

    cat file1 | sort -u > /tmp/one_list
cat file2 | sort -u > /tmp/two_list
comm -12 /tmp/one_list /tmp/two_list #gives intersection
comm -23 /tmp/one_list /tmp/two_list #gives lines only in file1 and not in file2
comm -13 /tmp/one_list /tmp/two_list #gives lines only in file2 and not in file1
